#include <SoftwareSerial.h>
SoftwareSerial bt(10, 11); // RX + Resistor, TX
const int ledPin = 13;
void setup() {
  pinMode(ledPin, OUTPUT);
  digitalWrite(ledPin, LOW);
  Serial.begin(9600);
  bt.begin(9600);   // default HC-06
}
void loop() {
  if (bt.available()) {
    char cmd = bt.read();
    if (cmd == '1') {
      digitalWrite(ledPin, HIGH);
      bt.println("LED ON");
    }
    if (cmd == '0') {
      digitalWrite(ledPin, LOW);
      bt.println("LED OFF");
    }
  }
}
